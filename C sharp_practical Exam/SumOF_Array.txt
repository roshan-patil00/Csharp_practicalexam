class SumOF_Array
{
     public static void Main(string[]arg)
        {
            int[] arr = new int[5];
            for(int i=0;i<5;i++)
            {
                Console.WriteLine("Enter number");
                arr[i] = int.Parse(Console.ReadLine());
            }

            int sum = 0;
            for(int i=0;i<5;i++)
            {
                sum = sum + arr[i];
            }
          Console.WriteLine("Sum of array is : " + sum);
        }
    }
}
